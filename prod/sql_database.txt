use LindeDataLogger;

CREATE TABLE DataLogger1
( id BIGINT IDENTITY PRIMARY KEY,
  site_id VARCHAR(40) NOT NULL,
  site_name VARCHAR(40),
  information VARCHAR(1000),
  created_at DATETIME,
  tier1 FLOAT,
  tier2 FLOAT,
  note VARCHAR(50) null
);

CREATE TABLE DataLogger2
( id BIGINT IDENTITY PRIMARY KEY,
  site_id VARCHAR(40) NOT NULL,
  site_name VARCHAR(40),
  avrflow FLOAT,
  tier1 FLOAT,
  tier2 FLOAT,
  created_at DATETIME
);


CREATE TABLE DataLoggerStatus
( id BIGINT IDENTITY PRIMARY KEY,
  site_id VARCHAR(40) NOT NULL,
  site_name VARCHAR(40),
  is_connect BIT,
  created_at DATETIME
);
